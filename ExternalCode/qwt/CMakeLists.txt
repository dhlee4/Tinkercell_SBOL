# THIS CMAKEFILE IS NO LONGER USED BECAUSE
# QWT SOURCE IS INCLUDED IN TINKERCELLCORE

IF( QT4_FOUND AND QT_VERSION_MINOR GREATER 5)

SET(LIBRARY_OUTPUT_PATH ${EXECUTABLE_OUTPUT_PATH})

FILE( GLOB QWT_SRC ${TINKERCELL_SOURCE_DIR}/ExternalCode/qwt/*.cpp )
FILE( GLOB QWT_HDRS ${TINKERCELL_SOURCE_DIR}/ExternalCode/qwt/*.h )
QT4_WRAP_CPP( QWT_MOC ${QWT_HDRS} )

ADD_DEFINITIONS(-DQWT_DLL -DQWT_MAKEDLL)

ADD_LIBRARY( qwt
#  STATIC
  SHARED
  ${QWT_SRC}
  ${QWT_MOC}
  )

TARGET_LINK_LIBRARIES( qwt
  qwt
  ${QT_LIBRARIES}
  )

IF ( WIN32 )
  INSTALL(TARGETS qwt DESTINATION ${TINKERCELL_PACKAGE_ROOT_DIR})
ELSE ( WIN32 )
  INSTALL(TARGETS qwt LIBRARY DESTINATION ${TINKERCELL_PACKAGE_ROOT_DIR})
ENDIF( WIN32 )

ENDIF( QT4_FOUND AND QT_VERSION_MINOR GREATER 5)
